--three level logic(true,false,null)
select lastname,firstname,region
from dbo.Employees
where region is null;

--And requires both condition are true
select lastname,city,postalcode
from dbo.Employees
where city ='seattle' AND postalcode like '9%'

--OR used one condition to be true
select lastname,city,postalcode
from dbo.employees
where city='seatle' OR postalcode like '9%'

--not negates expression
select lastname,city,postalcode
from dbo.employees
where city NOT like 'london' 

--operator procedure(AND,OR,NOT)
select lastname,city,postalcode
from dbo.Employees
where lastname like '%s%'
AND city not like 'seattle';

--cross join
select * from customer c cross join orderss o

--cross join differently
select * from customer ,orderss

--cross join with condition(inner join for a)
select * from customer c cross join orderss o where c.customerid=o.customerid

--a simple inner join
select * from  customer  join orderss on customer.customerid=orderss.customerid

-- another way to inner join without join keyword
select * from customer,orderss where customer.customerid=orderss.customerid

--another way to inner join
select * from customer c inner join orderss o on c.customerid=o.customerid
